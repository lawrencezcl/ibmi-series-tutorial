### 7.1 第一个项目：ERC-20代币发行

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.19;

import "@openzeppelin/contracts/token/ERC20/ERC20.sol";
import "@openzeppelin/contracts/access/Ownable.sol";

/**
 * @title MyToken
 * @dev 一个简单的ERC-20代币合约
 */
contract MyToken is ERC20, Ownable {
    uint256 public constant MAX_SUPPLY = 1000000 * 10**18; // 100万代币
    
    constructor() ERC20("MyToken", "MTK") {
        // 铸造初始供应量给合约部署者
        _mint(msg.sender, 100000 * 10**decimals()); // 10万代币
    }
    
    /**
     * @dev 铸造新代币（仅合约所有者）
     */
    function mint(address to, uint256 amount) public onlyOwner {
        require(totalSupply() + amount <= MAX_SUPPLY, "Exceeds max supply");
        _mint(to, amount);
    }
    
    /**
     * @dev 销毁代币
     */
    function burn(uint256 amount) public {
        _burn(msg.sender, amount);
    }
}
```

### 7.2 第二个项目：简单DEX (去中心化交易所)

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.19;

import "@openzeppelin/contracts/token/ERC20/IERC20.sol";
import "@openzeppelin/contracts/security/ReentrancyGuard.sol";

/**
 * @title SimpleDEX
 * @dev 基于AMM模式的简单去中心化交易所
 */
contract SimpleDEX is ReentrancyGuard {
    IERC20 public tokenA;
    IERC20 public tokenB;
    
    uint256 public reserveA;
    uint256 public reserveB;
    uint256 public totalSupply;
    mapping(address => uint256) public balanceOf;
    
    // 流动性提供者添加流动性事件
    event LiquidityAdded(address indexed provider, uint256 amountA, uint256 amountB);
    // 交换事件
    event Swap(address indexed user, address tokenIn, uint256 amountIn, uint256 amountOut);
    
    constructor(address _tokenA, address _tokenB) {
        tokenA = IERC20(_tokenA);
        tokenB = IERC20(_tokenB);
    }
    
    /**
     * @dev 添加流动性
     */
    function addLiquidity(uint256 amountA, uint256 amountB) external nonReentrant {
        require(amountA > 0 && amountB > 0, "Invalid amounts");
        
        // 转账代币到合约
        tokenA.transferFrom(msg.sender, address(this), amountA);
        tokenB.transferFrom(msg.sender, address(this), amountB);
        
        // 计算LP代币数量
        uint256 liquidity;
        if (totalSupply == 0) {
            liquidity = sqrt(amountA * amountB);
        } else {
            liquidity = min(
                (amountA * totalSupply) / reserveA,
                (amountB * totalSupply) / reserveB
            );
        }
        
        require(liquidity > 0, "Insufficient liquidity minted");
        
        // 铸造LP代币
        balanceOf[msg.sender] += liquidity;
        totalSupply += liquidity;
        
        // 更新储备
        reserveA += amountA;
        reserveB += amountB;
        
        emit LiquidityAdded(msg.sender, amountA, amountB);
    }
    
    /**
     * @dev 交换代币 (A -> B)
     * 使用恒定乘积公式: x * y = k
     */
    function swapAForB(uint256 amountAIn) external nonReentrant {
        require(amountAIn > 0, "Invalid input amount");
        require(reserveA > 0 && reserveB > 0, "Insufficient liquidity");
        
        // 计算输出金额 (考虑0.3%手续费)
        uint256 amountAInWithFee = (amountAIn * 997) / 1000;
        uint256 amountBOut = (reserveB * amountAInWithFee) / (reserveA + amountAInWithFee);
        
        require(amountBOut > 0, "Insufficient output amount");
        require(amountBOut < reserveB, "Output exceeds reserve");
        
        // 执行转账
        tokenA.transferFrom(msg.sender, address(this), amountAIn);
        tokenB.transfer(msg.sender, amountBOut);
        
        // 更新储备
        reserveA += amountAIn;
        reserveB -= amountBOut;
        
        emit Swap(msg.sender, address(tokenA), amountAIn, amountBOut);
    }
    
    // 辅助函数
    function sqrt(uint256 y) internal pure returns (uint256 z) {
        if (y > 3) {
            z = y;
            uint256 x = y / 2 + 1;
            while (x < z) {
                z = x;
                x = (y / x + x) / 2;
            }
        } else if (y != 0) {
            z = 1;
        }
    }
    
    function min(uint256 a, uint256 b) internal pure returns (uint256) {
        return a < b ? a : b;
    }
}
```

### 8.2 常见攻击类型

| 攻击类型 | 描述 | 防护措施 |
|---------|------|---------|
| **重入攻击** | 恶意合约回调重新进入 | Checks-Effects-Interactions |
| **整数溢出** | 数值计算溢出 | 使用0.8+或使用SafeMath |
| **访问控制** | 未授权函数调用 | 修饰器、权限检查 |
| **前置交易** | 矿工/套利者抢跑 | 使用commit-reveal模式 |
| **预言机操纵** | 价格数据源被攻击 | 使用去中心化预言机 |
| **拒绝服务** | Gas耗尽或逻辑卡住 | Gas限制检查、逻辑优化 |
